# ğŸŸ© **Now Controller will dynamically generate FORM (without Blade)**

```php
public function showDynamicForm()
{
    $json_path = storage_path('app/public/form_fields.json');
    $formData = json_decode(file_get_contents($json_path), true);

    $html = '<form method="POST" action="/submit-form">';
    $html .= csrf_field();

    foreach ($formData['form']['fields'] as $field) {

        $label = $this->translate($field['label'], 'ur');
        $html .= "<label>{$label}</label><br>";

        // Radio buttons
        if ($field['type'] === 'radio') {
            foreach ($field['options'] as $option) {
                $translatedOption = $this->translate($option, 'ur');
                $html .= "<input type='radio' name='{$field['key']}' value='{$option}'> {$translatedOption}<br>";
            }
        }
        else {
            // Normal inputs
            $html .= "<input type='{$field['type']}' name='{$field['key']}'><br><br>";
        }
    }

    $html .= "<button type='submit'>Submit</button>";
    $html .= "</form>";

    return $html;
}
```
